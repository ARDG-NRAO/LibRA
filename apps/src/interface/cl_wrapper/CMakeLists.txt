
set (CMAKE_CXX_STANDARD 11)
set(OPT "-g -O2 ")
message("Apps_BUILD_TESTS is: ${Apps_BUILD_TESTS}")
set(WRAPPER_LIST roadrunner hummbee mssplit subms tableinfo wabbler coyote)
foreach(WRAPPER IN LISTS WRAPPER_LIST)
	#message("Building CLWrapper: ${WRAPPER}")
	if(Apps_BUILD_TESTS)
	  #message("Building library: ${WRAPPER}")
	  add_library(${WRAPPER} ${WRAPPER}.cc)
	else()
	  #message("Building binary: ${WRAPPER}")
	  add_executable(${WRAPPER} ${WRAPPER}.cc)
	endif()
	target_link_libraries(${WRAPPER} PRIVATE ${APP_LINK_LIBRARIES})
	install(TARGETS ${WRAPPER} DESTINATION "${CMAKE_SOURCE_DIR}/install") 
endforeach()
